001package locationmapping;
002
003import processing.core.PGraphics;
004import de.fhpotsdam.unfolding.geo.Location;
005
006public class MarkerSMS extends StandardMarker {
007
008        /**
009        * Konstruktor für MarkerSMS Objekte
010        *
011        * @param location Ort fuer den Marker erstellt werden soll
012        * @return neues Objekt vom Typ MarkerSMS
013        */
014        public MarkerSMS(Location location){
015                super(location);
016        }
017        
018        /**
019        * Konstruktor für MarkerSMS Objekte
020        *
021        * @param trackpoint Trackpoint fuer den Marker erstellt werden soll
022        * @return neues Objekt vom Typ MarkerSMS
023        */
024        public MarkerSMS(Trackpoint trackpoint) {
025            super(trackpoint.getLocation());
026        }
027                
028        /**
029        * Zeichnet Marker
030        * @param pg Objekt das gezeichnet werden soll
031        * @param x X-Koordinate
032        * @param y Y-Koordinate
033        */
034        public void draw(PGraphics pg, float x, float y){
035                if (!this.isHidden()){
036                        pg.pushStyle();
037                    pg.noStroke();
038                    pg.fill(hsb_h, hsb_s, hsb_b, transparency);  // Farbe sowie sichtbarkeit
039                    pg.ellipse(x, y, size, size);  // Form
040                    pg.stroke(0, 0, 0, 200);  //Zeichnung des Symbols für die obere Ecke
041                    pg.rect(x + size/2 , y - size ,  size, size / 2);  //Umschlag
042
043                    //Setzt die beiden Linien
044                    pg.line(x+ size/2 , y - size , x + size , y - size * 0.75f );
045                    pg.line(x + size , y - size * 0.75f , x + 1.5f * size , y - size  );
046                    pg.popStyle();
047                }
048        }
049}
