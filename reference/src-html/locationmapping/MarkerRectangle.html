001package locationmapping;
002
003import processing.core.PGraphics;
004import de.fhpotsdam.unfolding.geo.Location;
005import de.fhpotsdam.unfolding.marker.SimplePointMarker;
006
007public class MarkerRectangle extends StandardMarker {
008        /**
009        * Konstruktor für MarkerRectangle Objekte
010        *
011        * @param location Ort fuer den Marker erstellt werden soll
012        * @return neues Objekt vom Typ MarkerRectangle
013        */
014        public MarkerRectangle(Location location){
015                super(location);
016        }
017        /**
018        * Konstruktor für MarkerRectangle Objekte
019        *
020        * @param trackpoint Trackpoint fuer den Marker erstellt werden soll
021        * @return neues Objekt vom Typ MarkerRectangle
022        */
023        public MarkerRectangle(Trackpoint trackpoint) {
024            super(trackpoint.getLocation());
025          }
026        /**
027        * Zeichnet Marker
028        * @param pg Objekt das gezeichnet werden soll
029        * @param x X-Koordinate
030        * @param y Y-Koordinate
031        */
032        public void draw(PGraphics pg, float x, float y){
033                if (!this.isHidden()){
034                        pg.pushStyle();
035                        pg.stroke(hsb_h, hsb_s, hsb_b, transparency);  // kein Rand
036                        pg.strokeWeight(2);
037                        pg.fill(hsb_h, hsb_s, hsb_b, transparency);  // Farbe sowie sichtbarkeit
038                        pg.rect(x, y, size, size);  // Form: Rechteck
039                        pg.popStyle();
040            return;
041                }
042        }
043}
