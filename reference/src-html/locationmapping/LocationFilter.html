001package locationmapping;
002
003import java.util.*;
004
005import de.fhpotsdam.unfolding.geo.*;
006
007
008
009public class LocationFilter extends Filter {
010    /**
011     * Mindestfrequenz für den Frequenzfilter
012     */
013    int minfrequency = 0;
014    /**
015    * Location, nach der gefiltert wird
016    */
017    Location location;
018    /**
019    * Radius um Location, die im Radiusfilter genutzt werden
020    */
021    int radius = 0;
022    /**
023    * Service, der im Servicefilter gefiltert wird (SMS, Internet, GPRS)
024    */
025    String service;
026
027
028    /**
029    * Setzt die Mindesthaeufigkeit
030    * @param minf Mindesthaeufigkeit
031    */
032    public void setMinFrequency(int minf){
033        minfrequency = minf;
034    }
035
036    /**
037    * Setzt den Radius
038    *
039    * @param radius Radius, um den gefiltert wird
040    */
041    public void setRadius(int radius){
042        this.radius = radius;
043    }
044
045    /**
046    * Setzt die Location
047    *
048    * @param location Ort, um den gefiltert wird
049    */
050    public void setLocation(Location location){
051        this.location = location;
052    }
053
054    /**
055    * Setzt den Service
056    *
057    * @param service Service, nach dem gefiltert werden soll
058    */
059    public void setService(String service){
060        this.service = service;
061    }
062
063    // zeichne Trackpoint nur, wenn beide innerhalb des Radius waren
064    public TrackpointList compareLocation(TrackpointList eingabetpl1, TrackpointList eingabetpl2, int radius){
065        TrackpointList newtpl = new TrackpointList();
066        for ( Trackpoint tp1 : eingabetpl1 ){
067           for ( Trackpoint tp2 : eingabetpl2 ){
068                if ( tp1.locationDistanceTo(tp2.getLocation()) <= radius ){
069                    newtpl.add(tp1);
070                    break;
071                }
072            }
073        }
074        return newtpl;
075    }
076
077    //zeichne Trackpoint nur, wenn Ort und Zeit übereinstimmen
078    public TrackpointList compareLocationAndTime(TrackpointList eingabetpl1, TrackpointList eingabetpl2, int radius, int minutes){
079        TrackpointList newtpl = new TrackpointList();
080        for (Trackpoint tp1 : eingabetpl1){
081            for (Trackpoint tp2 : eingabetpl2){
082                if (  (tp1.locationDistanceTo(tp2.getLocation()) <= radius) && (tp1.locationDistanceTo(tp2) < minutes * 60) ){
083                    newtpl.add(tp1);
084                 break;
085                }
086            }
087        }
088        return newtpl;
089    }
090
091
092    /**
093    * Filtert nach Frequenz
094    * @param trackpointlist Liste, die gefiltert werden soll
095    */
096    private void frequencyFilter(TrackpointList trackpointlist){
097        for(Trackpoint tp : trackpointlist){
098            if(trackpointlist.getFrequency(tp) < minfrequency)
099                tp.setVisible(false);
100        }
101    }
102
103    /**
104    * Filtert nach Location und Radius
105    * @param trackpointlist Liste, die gefiltert werden soll
106    */
107    private void locationFilter(TrackpointList trackpointlist){
108        for( Trackpoint tp : trackpointlist ){
109            if (tp.locationDistanceTo(location) > radius)
110                tp.setVisible(false);
111        }
112    }
113
114    /**
115    * Filtert nach Service
116    * @param trackpointlist Liste, die gefiltert werden soll
117    */
118    private void serviceFilter(TrackpointList trackpointlist){
119        for(Trackpoint tp : trackpointlist){
120            if(!(tp.getService().equals(service))){
121                tp.setVisible(false);
122            }
123        }
124    }
125
126
127    /**
128     * apply Methode, die je nachdem, wie die Attribute gesetzt wurden filtert
129     * Es wird ueberprueft, ob und welche Attribute gesetzt wurden, so dass bestimmte Filter aufgerufen werden
130     *
131     * @param trackpointlist Die TrackpointList, die gefiltert wird
132     * @return gibt eine TrackpointList zurueck, in der die visible Eigenschaft veraendert wurde
133    */
134    public TrackpointList apply(TrackpointList trackpointlist) {
135
136     // Frequenzfilter
137     if (minfrequency > 1) {
138        frequencyFilter(trackpointlist);
139     }
140
141     // Radiusfilter
142     if (radius != 0 && location != null) {
143        locationFilter(trackpointlist);
144     }
145
146     //Servicefilter
147     if (service != null) {
148        serviceFilter(trackpointlist);
149     }
150
151     //ueberschreiben der Rauszufilternden Punkte
152     TrackpointList filteredtpl = new TrackpointList();
153     for(Trackpoint tp : trackpointlist){
154         if(tp.getVisible() == true){
155             filteredtpl.add(tp);
156         }else{
157             tp.setVisible(true);
158         }
159     }
160     return filteredtpl;
161    }
162}
