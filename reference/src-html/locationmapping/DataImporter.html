001package locationmapping;

003import org.joda.time.*;

005import processing.core.PApplet;

008import de.fhpotsdam.unfolding.geo.Location;

010public class DataImporter {
011    public static final int UNIX = 0;
012    public static final int ISO8601 = 1;
013    public static final int EXPONENT_APPLE = 2;
014    public static final int MDY_DATETIME = 3;
015
016    /**
017    * laufende Processing Applet Instanz
018    */
019    PApplet app;
020    /**
021    * Mindestabstand zwischen zwei einzulesenden Zeitpunkten in Sekunden (default = "60")
022    */
023    private int minTimeDistance = 60;
024    /**
025    * Maximale Anzahl einzulesender Datenpunkte, -1 = alle
026    */
027    private int maxImportSize = -1;
028    /**
029    * Genauigkeit in der Ortskoordinaten eingelesen werden sollen in Grad (default = "0.0001")
030    */
031    private double accuracy = 0.0001;
032
033    /**
034    * Erzeugt neues DataImporter Objekt im Applet
035    *
036    * @param app laufendes Processing Applet in das importiert werden soll
037    */
038    public DataImporter(PApplet app) {
039        this.app = app;
040    }
041
042    /**
043    * Setzt maximale Anzahl der einzulesenden Datenpunkte
044    *
045    * @param maxImportSize maximale Anzahl einzulesender Datenpunkte
046    */
047    public void setMaxImportSize(int maxImportSize){
048        this.maxImportSize = maxImportSize;
049    }
050
051    /**
052    * Setzt Mindestzeitabstand zwischen zwei Zeitpunkten fest
053    *
054    * @param minTimeDistance Mindestabstand zwischen zwei Zeitpunkten in Sekunden
055    */
056    public void setMinTimeDistance(int minTimeDistance){
057        this.minTimeDistance = minTimeDistance;
058    }
059
060    /**
061    * Setzt Genauigkeit fuer Datenimporter
062    *
063    * @param accuracy Zahl zu derem Vielfachen gerundet wird in Grad
064    */
065    public void setAccuracy(double accuracy){
066        this.accuracy = accuracy;
067    }
068
069    /**
070    * Rundet Zahl auf ganzzahliges vielfaches eines Inkrements
071    *
072    * @param number zu rundende Zahl
073    * @param increment Inkrement zu dessen Vielfachem gerundet werden soll
074    */
075    double round(double number, double increment){
076        return ((double) Math.round(number/increment))*increment;
077    }
078
079    /**
080     * Importiert Daten
081     *
082     * @param filename Name der zu importierenden Datei im data Ordner
083     * @param id (default = 1) Identifikationsnummer der Datensatzes
084     * @return Trackpointliste mit Datenpunkten
085     * @throw RuntimeException, falls Dateiformat unbekannt
086     */
087    public TrackpointList load(String filename, int id, int timeFormat){
088        // parse Dateiendung
089        String extension = "";
090        int i = filename.lastIndexOf('.');
091        if ( i > 0 )
092            extension = filename.substring(i+1);
093
094        if ( extension.equals("csv") || extension.equals("tsv") ){
095            return loadSpreadsheet(filename, id, timeFormat);
096        }
097        if ( extension.equals("json") ){
098            return loadGoogleJSON(filename, id);
099        }
100        throw new RuntimeException("Unknown File Format");
101    }
102
103
104    /**
105     * Importiert von Google exportierte JSON Daten
106     *
107     * @param filename Name der zu importierenden Datei im data Ordner
108     * @param id(default = 1) Identifikationsnummer der Datensatzes
109     * @return Trackpointliste mit Datenpunkten
110     */
111    public TrackpointList loadGoogleJSON(String filename, int id){
112        TrackpointList trackpointList = new TrackpointList();
113
114        // Extrahiere Array aus Daten
115        JSONObject wrapperObject = app.loadJSONObject(filename);
116        JSONArray data = wrapperObject.getJSONArray("locations");
117
118        // speichert letzten Zeitstempel fuer Ueberpruefung der minTimeDistance
119        long lastTimestamp = 0;
120
121        // Laufe ueber Array mit Daten
122        int counter = 0;
123        for ( int i=0; i<data.size(); i++ ) {
124            JSONObject row = data.getJSONObject(i);
125
126            // Brich Import ab, falls maxImportSize ueberschritten
127            if ( counter > this.maxImportSize && this.maxImportSize > 0 )
128                break;
129
130            // Zeitstempel der Zeile
131            long timestamp = Long.parseLong(row.getString("timestampMs"));
132
133            // ignoriere Zeile, falls Zeitunterschied kleiner als minTimeDistance
134            if ( Math.abs(lastTimestamp - timestamp) > this.minTimeDistance*1000 ){
135                lastTimestamp = timestamp;
136                counter++;
137
138                // berechne float mit Laengengrad der Zeile
139                String longitudeString = Long.toString(row.getLong("longitudeE7"));
140                longitudeString = longitudeString.substring(0, 2) + "." + longitudeString.substring(2);
141                float longitude = Float.parseFloat(longitudeString);
142
143                // berechne float mit Breitengrad der Zeile
144                String latitudeString = Long.toString(row.getLong("latitudeE7"));
145                latitudeString = latitudeString.substring(0, 2) + "." + latitudeString.substring(2);
146                float latitude = Float.parseFloat(latitudeString);
147
148                // erstelle Trackpoint und füge ihn zu Liste hinzu
149                Location location = new Location(round(latitude, this.accuracy), round(longitude, this.accuracy));
150                Trackpoint trackpoint  = new Trackpoint(new DateTime(timestamp), location, id, "GPS");
151                trackpointList.add(trackpoint);
152                // System.out.println(counter + ": " + trackpoint);
153            }// end if
154        }// end for
155        return trackpointList;
156    }
157
158    /**
159     * Importiert CSV und TSV Daten von Tabellen mit Headern Timestamp/DateTime, Longitude, Latitude (evtl. Service)
160     *
161     * @param filename Name der zu importierenden Datei im data Ordner
162     * @param id (default = 7) Identifikationsnummer der Datensatzes
163     * @return Trackpointlist mit Datenpunkten
164     */
165    public TrackpointList loadSpreadsheet(String filename, int id, int timeFormat){
166        TrackpointList trackpointList = new TrackpointList();
167
168        // Extrahiere Array aus Daten
169        Table data = app.loadTable(filename, "header");
170
171        // speichert letzten Zeitstempel fuer Ueberpruefung der minTimeDistance
172        DateTime lastTimestamp = new DateTime(0);
173
174        // Laufe ueber Array mit Daten
175        int counter = 0;
176        for ( TableRow row : data.rows() ) {
177
178            // Brich Import ab, falls maxImportSize ueberschritten
179            if ( counter > this.maxImportSize && this.maxImportSize > 0 )
180                break;
181
182            // Zeitstempel der Zeile
183            DateTime timestamp = new DateTime(0);
184
185            if ( timeFormat == MDY_DATETIME )
186                timestamp = parseDateTimeString(row.getString("DateTime"));
187            else if ( timeFormat == EXPONENT_APPLE )
188                timestamp = parseTimestampString(row.getString("Timestamp"));
189            else if ( timeFormat == UNIX )
190                timestamp = new DateTime(row.getLong("Timestamp"));
191            else if ( timeFormat == ISO8601 )
192                timestamp = new DateTime(row.getString("DateTime"));
193            else
194                throw new RuntimeException("incompatible DateTime Format");
195
196
197            // ignoriere Zeile, falls Zeitunterschied kleiner als minTimeDistance
198            if ( Seconds.secondsBetween(lastTimestamp, timestamp).getSeconds() > this.minTimeDistance ){
199                lastTimestamp = timestamp;
200                counter++;
201
202                // Breitengrad und Längengrad der Zeile
203                float latitude = row.getFloat("Latitude");
204                float longitude = row.getFloat("Longitude");
205
206                // Service der Zeile
207                String service = "";
208                try {
209                    service = row.getString("Service");
210                } catch(IllegalArgumentException e) {}
211
212
213                // erstelle Trackpoint und füge ihn zu Liste hinzu
214                Location location = new Location(round(latitude, this.accuracy), round(longitude, this.accuracy));
215                Trackpoint trackpoint  = new Trackpoint(timestamp, location, id, service);
216                trackpointList.add(trackpoint);
217                // System.out.println(counter + ": " + trackpoint);
218            }
219        }
220        return trackpointList;
221    }

223    /**
224     * Wandelt Datenstring in Zeitformat um
225     *
226     * @param dateTimeString String mit Datum und Zeit
227     * @return aus dem String erstelltes DateTime Objekt
228     */
229    DateTime parseDateTimeString(String dateTimeString){
230        int month =  Integer.parseInt(dateTimeString.substring(0,2));
231        int day =  Integer.parseInt(dateTimeString.substring(3,5));
232        int year =  Integer.parseInt(dateTimeString.substring(6,10));
233        int hour = Integer.parseInt(dateTimeString.substring(11,13));
234        int minute = Integer.parseInt(dateTimeString.substring(14,16));
235        int second = Integer.parseInt(dateTimeString.substring(17,19));
236        DateTimeZone zone = DateTimeZone.forID("Europe/Berlin");
237        return new DateTime(year, month, day, hour, minute, second, zone);
238    }
239
240    /**
241     * Wandelt Timestamp in Zeitformat um
242     *
243     * @param timestampString String mit Datum und Zeit
244     * @return aus dem String erstelltes DateTime Objekt
245     */
246    DateTime parseTimestampString(String timestampString){
247        long timestamp = Long.parseLong(timestampString.substring(0, 1) + timestampString.substring(2, 10)) + 978285600;
248        return new DateTime(timestamp);
249    }
