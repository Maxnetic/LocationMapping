001package locationmapping;
002
003import processing.core.PGraphics;
004import de.fhpotsdam.unfolding.geo.Location;
005
006public class MarkerAnruf extends StandardMarker {
007        /**
008        * Konstruktor für MarkerAnruf Objekte
009        *
010        * @param location Ort fuer den Marker erstellt werden soll
011        * @return neues Objekt vom Typ MarkerAnruf
012        */
013        public MarkerAnruf(Location location){
014                super(location);
015        }
016        /**
017        * Konstruktor für MarkerAnruf Objekte
018        *
019        * @param trackpoint Trackpoint fuer den Marker erstellt werden soll
020        * @return neues Objekt vom Typ MarkerAnruf
021        */
022        public MarkerAnruf(Trackpoint trackpoint) {
023            super(trackpoint.getLocation());
024        }
025        /**
026        * Zeichnet Marker
027        * @param pg Objekt das gezeichnet werden soll
028        * @param x X-Koordinate
029        * @param y Y-Koordinate
030        */      
031        public void draw(PGraphics pg, float x, float y){
032                if (!this.isHidden()){
033                        pg.pushStyle();
034                    pg.noStroke();
035                    pg.fill(hsb_h, hsb_s, hsb_b, transparency);  // Farbe sowie sichtbarkeit
036                    pg.ellipse(x, y, size, size);  // Zeichnet einen Kreis
037
038                    //Zeichnung des Symbols für die obere Ecke
039                    pg.stroke(0,0,0);
040                    pg.rect(x + size/2 , y - size ,  size * 0.4f, size / 2f);
041                    pg.rect(x + size * 0.74f , y - size * 1.24f ,  size/6f, size/4f);
042                    pg.rect(x + size * 0.57f, y - size * 0.93f, size * 0.25f, size * 0.2f);
043                    pg.popStyle();
044                }
045        }
046}
