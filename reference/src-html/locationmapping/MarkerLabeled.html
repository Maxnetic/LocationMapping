001package locationmapping;
002
003import processing.core.PGraphics;
004import de.fhpotsdam.unfolding.geo.Location;
005
006public class MarkerLabeled extends StandardMarker {
007        /**
008        * Konstruktor für MarkerLabeled Objekte
009        *
010        * @param location Ort fuer den Marker erstellt werden soll
011        * @return neues Objekt vom Typ MarkerLabeled
012        */
013        public MarkerLabeled(Location location){
014                super(location);
015        }
016        /**
017        * Konstruktor für MarkerLabeled Objekte
018        *
019        * @param trackpoint Trackpoint fuer den Marker erstellt werden soll
020        * @return neues Objekt vom Typ MarkerLabeled
021        */
022        public MarkerLabeled(Trackpoint trackpoint) {
023            super(trackpoint.getLocation());
024        }
025        /**
026        * Zeichnet Marker
027        * @param pg Objekt das gezeichnet werden soll
028        * @param x X-Koordinate
029        * @param y Y-Koordinate
030        */
031        public void draw(PGraphics pg, float x, float y){
032                if (!this.isHidden()){
033                        pg.pushStyle();
034                    pg.stroke(hsb_h,hsb_s,hsb_b);  // kein Rand
035                    pg.strokeWeight(2);
036                    pg.fill(hsb_h, hsb_s, hsb_b, transparency);  // Farbe sowie sichtbarkeit
037                    pg.rect(x, y, size, size);  // Form: Rechteck
038                    pg.text(label, x + size + 5 , y + size/2);
039                    pg.textSize(textsize);
040                    pg.popStyle();
041                }
042        }
043}
