328    /**
329     * Hilfsfunktion, die einen String in ein Zeitobjekt umwandelt
330     *
331     * @param timeString Zeitstring im Format: HH:MM or HH
332     * @return gibt Zeitobjekt zurueck, das eingegebene Zeit repraesentiert
333     * @throws RuntimeException, falls String nicht geparsed werden kann
334     */
335    LocalTime parseTime(String timeString){
336        timeString = timeString.trim();
337        try {
338            int hour;
339            int minute = 0;
340            if ( timeString.contains(":") || timeString.contains("\\.") ){
341                String[] aux;
342                if ( timeString.contains(":") )
343                    aux = timeString.split(":");
344                else
345                    aux = timeString.split("\\.");
346                hour = Integer.parseInt(aux[0].replaceAll("[\\D]", ""));
347                minute = Integer.parseInt(aux[1].replaceAll("[\\D]", ""));
348            } else {
349                hour = Integer.parseInt(timeString.replaceAll("[\\D]", ""));
350            }
351            return new LocalTime(hour, minute);
352        } catch (Exception e) {
353            throw new RuntimeException("'" + timeString + "' not in parsable format H:MM or HH");
354        }
355    }
356
357    /**
358     * Hilfsfunktion, die einen String in ein Datumsobjekt umwandelt
359     *
360     * @param timeIntervalString Zeitintervalstring im Format: HH:MM - HH:MM oder ohne Minuten
361     * @return gibt Zeitarray zurueck, welches eingegebenes Zeitinterval reprasesentiert, start in [0], ende in [1]
362     * @throws RuntimeException, falls String nicht geparsed werden kann
363     */
364    LocalTime[] parseTimeInterval(String timeIntervalString){
365        try {
366            LocalTime[] interval = new LocalTime[2];
367            // Eingabe an "-" splitten
368            String[] aux = timeIntervalString.split("-");
369            // Start und Enddatum parsen
370            interval[0] = parseTime(aux[0].trim());
371            interval[1] = parseTime(aux[1].trim());
372            return interval;
373        } catch(Exception e){
374            throw new RuntimeException("'" + timeIntervalString + "' not in parsable format HH:MM - HH");
375        }
376    }

378    /**
379     * Hilfsfunktion, die einen String in ein Datumsobjekt umwandelt
380     *
381     * @param dateString Datumsstring im Format: YYYY/MM/DD oder D.M.YY bzw DD.MM.YYYY
382     * @return gibt Datumsobjekt zurueck, der die eingegebene Zeit repraesentiert
383     * @throws RuntimeException, falls String nicht geparsed werden kann
384     */
385    LocalDate parseDate(String dateString){
386        dateString = dateString.trim();
387        try {
388            int year, month, day;
389            if ( dateString.contains(".") ){
390                String[] aux = dateString.split("\\.");
391                day = Integer.parseInt(aux[0].replaceAll("[\\D]", ""));
392                month = Integer.parseInt(aux[1].replaceAll("[\\D]", ""));
393                year = Integer.parseInt(aux[2].replaceAll("[\\D]", ""));
394                if ( year < 100 ){
395                    if ( year < (new DateTime().getYear()-2000) )
396                        year += 2000;
397                    else
398                        year += 1900;
399                }
400            } else {
401                year = Integer.parseInt(dateString.substring(0,4));
402                month = Integer.parseInt(dateString.substring(5,7));
403                day = Integer.parseInt(dateString.substring(8,10));
404            }
405            return new LocalDate(year, month, day);
406        } catch (Exception e) {
407            throw new RuntimeException("'" + dateString + "' not in parsable format YYYY-MM-DD or D.M.YY");
408        }
409    }

428        } catch(Exception e){
429            throw new RuntimeException("'" + dateIntervalString + "' not in parsable format YYYY/MM/DD-YYYY/MM/DD or D.M.YY-D.M.YY");
430        }

433    /**
434     * Setzt zu filternde Wochentage
435     *
436     * @param weekDayString Eingabestring mit Wochentagen, der Form "mo - dienstag, friday" (erkennt alles)
437     */
438    public DateTimeFilter setWeekDays(String weekDayString){
439        weekDayString.toLowerCase();
440
441        // Loesche alle Leerzeichen um -
442        weekDayString = weekDayString.replace(" - ", "-");
443        weekDayString = weekDayString.replace(" -", "-");
444        weekDayString = weekDayString.replace("- ", "-");
445
446        // Splitte String an allen ',' und ' '
447        ArrayList<String> splittedString = new ArrayList<String>();
448        String[] aux = weekDayString.split(",");
449        for ( String str1 : aux ) {
450            for ( String str2 : str1.split(" ") ){
451                if ( !str2.isEmpty() )
452                    splittedString.add(str2);
453            }
454        }
455
456        // Parse Tage aus String
457        for ( String str : splittedString ){
458            if ( str.contains("-") ){
459                // Parse Wochentag-Ranges aus String
460                aux = new String[2];
461                try {
462                    aux = str.split("-");
463                    int d = parseWeekDay(aux[0]);
464                    while( d != dateInc(parseWeekDay(aux[1])) ){
465                        this.weekDays[d] = true;
466                        d = dateInc(d);
467                    }
468                } catch(Exception e) {
469                    throw new RuntimeException("'" + str + "' isn't parsable");
470                }
471            } else {
472                // Parse einzelne Tage aus String
473                this.weekDays[parseWeekDay(str)] = true;
474            }
475        }
476        return this;
477    }

487    /**
488     * Incrementiert Wochentagszahl, beachtet So nach Sa
489     *
490     * @param d Zahl zwischen 0 und 6 fuer Wochentag
491     * @return Zahl zwischen 0 und 6 fuer folgenden Wochentag
492     */
493    private int dateInc(int d) {
494        if ( d > 0 && d < 7 )
495            return d+1;
496        if ( d == 7 )
497            return 1;
498        else
499            throw new RuntimeException(d + " doesn't represent a weekday 1-7");
500    }

502    /**
503     * Gibt Zahl des Wochentage aus
504     *
505     * @param weekDayString Wochentagname der Zeitvariable in deutsch ("montag") oder englisch ("monday"), ausgeschrieben oder mit zwei Buchstaben Abkuerzung ("mo")
506     * @return Wochentag als Zahl mo=1 bis so=7
507     */
508    public int parseWeekDay(String weekDayString){
509        weekDayString.toLowerCase();
510        switch(weekDayString){
511            case "montag":
512            case "monday":
513            case "mo":
514                return 1;
515            case "dienstag":
516            case "di":
517            case "tuesday":
518            case "tu":
519                return 2;
520            case "mittwoch":
521            case "mi":
522            case "wednesday":
523            case "we":
524                return 3;
525            case "donnerstag":
526            case "do":
527            case "thursday":
528            case "th":
529                return 4;
530            case "freitag":
531            case "friday":
532            case "fr":
533                return 5;
534            case "samstag":
535            case "saturday":
536            case "sa":
537                return 6;
538            case "sonntag":
539            case "so":
540            case "sunday":
541            case "su":
542                return 7;
543        }
544        throw new RuntimeException("'" + weekDayString + "' doesn't represent a weekday");
545    }

551    /**
552     * Filtert die Liste nach den angegebenen Zeitintervallen, falls welche angegben sind
553     */
554    private void filterByTime(){
555        for ( Iterator<Trackpoint> iter = filteredList.iterator(); iter.hasNext(); ){
556            Trackpoint trackpoint = iter.next();
557            if ( !containedInTimeIntervals(trackpoint) )
558                iter.remove();
559        }
560    }

562    /**
563     * Hilfsmethode um zu ueberpruefen, ob Trackpoint in einem der spezifizierten Zeitintervalle liegt
564     *
565     * @param trackpoint Trackpoint, der ueberprueft werden soll
566     * @return Wahrheitswert darueber, ob Trackpoint in einem der Filter Zeitintervalle liegt, falls keines spezifiziert ist immer whar
567     */
568    boolean containedInTimeIntervals(Trackpoint trackpoint){
569        boolean contained = false;
570        LocalTime tpTime = new LocalTime(trackpoint.getHour(), trackpoint.getMinute(), trackpoint.getSecond());
571        if ( !this.timeIntervals.isEmpty() ){
572            // Zeitintervalle gesetzte
573            for ( LocalTime start : this.timeIntervals.keySet() ){
574                LocalTime end = this.timeIntervals.get(start);
575                contained = tpTime.compareTo(start) > 0 && tpTime.compareTo(end) < 0;
576                if ( contained ) break;
577            }
578        } else if ( this.startTime != null && this.endTime != null ){
579            // Start- und Endzeit gesetzte
580            contained = tpTime.compareTo(this.startTime) > 0 && tpTime.compareTo(this.endTime) < 0;
581        } else {
582            // kein Zeitfilter gesetzt
583            contained = true;
584        }
585        return contained;
586    }

588    /**
589     * Initialisiert die filteredList und schreankt sie dabei evtl. auf gesetzte Zeitintervalle fest
590     *
591     * @param list Trackpointliste, die gefiltert werden soll
592     */
593    private void setFilteredList(TrackpointList list){
594        Iterator<Trackpoint> iter;
595
596        if ( !dateIntervals.isEmpty() ){
597            // Falls Intervalle gesetzt wurden, verwende diese zum Filtern
598            for ( Interval interval : dateIntervals ){
599                iter = list.iterator(interval.getStart());
600                while ( iter.hasNext() ){
601                    Trackpoint trackpoint = iter.next();
602                    if ( trackpoint.compareTimeTo(interval.getEnd()) > 0 )
603                        break;
604                    filteredList.add(trackpoint);
605                }
606            }
607        } else {
608            // Falls keine Intervalle gesetzt wurden nutze Startpunkt oder starte vorne in Liste
609            if ( this.startDate != null ){
610                if ( list.getFirst().compareTimeTo(this.startDate) > 0 )
611                    System.out.println("startdate of filter " + startDate + " before first entry in list " + list.getFirst().getTime());
612                iter = list.iterator(startDate);
613            } else {
614                iter = list.iterator();
615            }
616            while ( iter.hasNext() ) {
617                Trackpoint trackpoint = iter.next();
618                if ( this.endDate != null ){
619                    if ( trackpoint.compareTimeTo(endDate) > 0 )
620                        break;
621                }
622                filteredList.add(trackpoint);
623            }
624        }
625    }

630    private void filterByWeekday(){
631        if ( this.weekDaysSet() ){
632            for ( Iterator<Trackpoint> iter = filteredList.iterator(); iter.hasNext(); ){
633                Trackpoint trackpoint = iter.next();
634                if ( !weekDays[trackpoint.getDayOfWeek()] )
635                    iter.remove();
636            }
637        }

642     *
643     * @return Wahrheitswert ob Wochentagfilter gesetzt ist

646        for ( boolean b : this.weekDays ){
647            if ( b ) return true;

652     /**
653     * Filtert angegebene Trackpointliste und gibt neue gefilterte Liste zurueck
654     *
655     * @param list zu filternde Trackpointliste, wird nicht veraendert
656     * @return neue gefilterte Trackpointliste
