001package locationmapping;
002
003import java.util.List;
004
005import processing.core.PConstants;
006import processing.core.PGraphics;
007import de.fhpotsdam.unfolding.geo.*;
008import de.fhpotsdam.unfolding.marker.*;
009import de.fhpotsdam.unfolding.utils.MapPosition;
010
011
012
013public class ColoredLinesMarker extends SimpleLinesMarker {
014    /**
015    * Groesse des Markers (default == 30)
016    */
017    int size = 30;
018    /**
019    * Farbwert der HSB Farbe
020    */
021    int hsb_h = 0;
022    /**
023    * Saettigungswert der HSB Farbe
024    */
025    int hsb_s = 0;
026    /**
027    * Helligkeitswert der HSB Farbe
028    */
029    int hsb_b = 0;
030    /**
031    * Transparenz der Farbe des Markers
032    */
033    int transparency = 100;
034
035    /**
036    * Kosntruktor fuer ColoredLineMarker Objekte
037    */
038    public ColoredLinesMarker() {
039        super();
040    }
041
042
043    /**
044    * Konstruktor fuer ColoredLineMarker Objekte
045    *
046    * @param start Startpunkt
047    * @param end Endpunkt
048    * @return neues Objekt vom Typ ColoredLineMarker
049    */
050    public ColoredLinesMarker(Location start, Location end) {
051        super(start, end);
052    }
053
054
055    /**
056    * Konstruktor fuer ColoredLineMarker Objekte
057    *
058    * @param locations Liste aus der Line gezeichnet werden soll
059    * @return neues Objekt vom Typ ColoredLineMarker
060    */
061    public ColoredLinesMarker(List<Location> locations){
062        super(locations);
063    }
064    /**
065    * Setzt Tansparenz
066    *
067    * @param trans Transparenzwert
068    */
069    public void setTransparency(int trans){
070        transparency = trans;
071      }
072
073      /**
074      * Setzt Farbe des Markers
075      *
076      * @param hsb_h Farbwert
077      * @param hsb_s Saettigungswert
078      * @param hsb_b Helligkeitswert
079      */
080      public void setColor(int hsb_h, int hsb_s, int hsb_b){
081        this.hsb_h =hsb_h;
082        this.hsb_s = hsb_s;
083        this.hsb_b = hsb_b;
084      }
085
086      /**
087      * Setzt Farbe des Markers
088      * @param colorstr Farbe
089      */
090
091      public void setColor(String colorstr){
092        colorstr.toLowerCase();
093        if (colorstr.equals("rot")){
094            hsb_h = 0;
095            hsb_s = 99;
096            hsb_b = 99;
097        } else  if (colorstr.equals("blau")){
098            hsb_h = 240;
099            hsb_s = 99;
100            hsb_b = 99;
101        } else if (colorstr.equals("grün")){
102            hsb_h = 100;
103            hsb_s = 99;
104            hsb_b = 99;
105        } else if (colorstr.equals("gelb")){
106            hsb_h = 60;
107            hsb_s = 99;
108            hsb_b = 99;
109        } else if (colorstr.equals("grau")){
110            hsb_h = 0;
111            hsb_s = 1;
112            hsb_b = 60;
113        }else{
114        System.out.println("Die Farbe ist nicht in der Liste! Farbe mit HSB Codierung möglich!");
115        }
116
117      }
118
119    /**
120     * Zeichenmethode
121     */
122    public void draw(PGraphics pg, List<MapPosition> mapPositions){
123        if (!this.isHidden()){
124            if ( mapPositions.isEmpty() || isHidden() ){
125                return;
126            }
127            pg.pushStyle();
128            pg.noFill();
129            if ( isSelected() ){
130              pg.stroke(highlightColor);
131            } else {
132                pg.stroke(hsb_h, hsb_s, hsb_b, transparency);
133            }
134            pg.strokeWeight(strokeWeight);
135            pg.smooth();
136
137            pg.beginShape(PConstants.LINES);
138            MapPosition last = mapPositions.get(0);
139            for (int i = 1; i < mapPositions.size(); ++i) {
140                MapPosition mp = mapPositions.get(i);
141                pg.vertex(last.x, last.y);
142                pg.vertex(mp.x, mp.y);
143                last = mp;
144            }
145            pg.endShape();
146            pg.popStyle();
147        }
148    }
149}
