425    public boolean equals(Trackpoint trackpoint, long timeTolerance, double locationTolerance) {
426        return this.equalLocation(trackpoint, locationTolerance) && this.equalTime(trackpoint, timeTolerance);

430    /**
431    * Gibt Stringrepraesentation des Trackpoint zurueck
432    *
433    * @return Stringrepraesentation des Trackpoint
434    */
435    public String toString(){
436        String out = "Trackpoint(id " + String.format("%03d",this.id) + " | Location " +  this.location.getLat() + " " + this.location.getLon() + " | Timestamp " + this.time;
437        if ( !this.service.isEmpty() )
438            out += " | Service " + this.service;
439        if ( !this.label.isEmpty() )
440            out += " | Label '" + this.label + "'";
441        return out + ")";
442    }
