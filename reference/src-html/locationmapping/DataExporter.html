001package locationmapping;
002
003import processing.core.PApplet;
004import processing.data.*;
005
006public class DataExporter {
007    /**
008    * laufende Processing Applet Instanz
009    */
010    PApplet app;
011    /**
012    * Maximale Anzahl der zu exportierenden Daten, -1 = alle
013    */
014    private int maxExportSize = -1;
015
016    /**
017    * Erzeugt neues DataExporter Objekt im Applet
018    *
019    * @param app laufendes Processing Applet in das importiert werden soll
020    */
021    public DataExporter(PApplet app) {
022        this.app = app;
023    }
024
025    /**
026    * Setzt maximale Anzahl der zu exportierenden Datenpunkte
027    *
028    * @param maxExportSize maximale Anzahl einzulesender Datenpunkte
029    */
030    public void setMaxExportSize(int maxExportSize){
031        this.maxExportSize = maxExportSize;
032    }
033
034    public boolean write(TrackpointList trackpointlist, String filename){
035        try {
036            //Anlegen der Tabelle
037            Table table = new Table();
038            table.addColumn("Timestamp");
039            table.addColumn("Longitude");
040            table.addColumn("Latitude");
041            table.addColumn("Service");
042
043            //Durchgehen der TrackpointList und Ã¼berschreiben
044            int counter = 0;
045            for ( Trackpoint trackpoint : trackpointlist ){
046                if ( counter > this.maxExportSize && this.maxExportSize > 0 )
047                    break;
048                TableRow newRow = table.addRow();
049                newRow.setLong("Timestamp", trackpoint.getTimestamp());
050                newRow.setFloat("Longitude", trackpoint.getLongitude());
051                newRow.setFloat("Latitude", trackpoint.getLatitude());
052                newRow.setString("Service", trackpoint.getService());
053                counter++;
054            }
055            app.saveTable(table, "data/" + filename + ".csv");
056            return true;
057        } catch(Exception e) {
058            return false;
059        }
060    }
061
062
063}
