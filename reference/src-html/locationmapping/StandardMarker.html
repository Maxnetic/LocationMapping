001package locationmapping;
002
003import de.fhpotsdam.unfolding.marker.*;
004import de.fhpotsdam.unfolding.geo.*;
005import processing.core.PGraphics;
006
007
008public class StandardMarker extends SimplePointMarker {
009 /**
010 * Größe des Markers
011 */
012  int size = 30;
013  /**
014  * Farbwert der HSB Farbe
015  */
016  int hsb_h = 0;
017  /**
018  * Sättigungswert der HSB Farbe
019  */
020  int hsb_s = 0;
021  /**
022  * Helligkeitswert der HSB Farbe
023  */
024  int hsb_b = 0;
025  /**
026  * Transparenz der Farbe des Markers
027  */
028  int transparency = 100;
029  /**
030  * Text für die Beschriftung
031  */
032  String label = null;
033  /**
034  * Textgröße der Beschriftung
035  */
036  int textsize = 12;
037
038  /**
039  * Konstruktor fuer StandardMarker Objekte
040  * @param location Ortsangabe des Markers
041  * @return neues Objekt vom Typ Marker
042  */
043  public StandardMarker(Location location) {
044    super(location);
045  }
046
047  /**
048   * Konstruktor fuer StandardMarker Objekte
049   * @param trackpoint Trackpoint aus dem Marker gezeichnet werden soll
050   * @return neues Objekt von Typ Marker
051   */
052  public StandardMarker(Trackpoint trackpoint) {
053    super(trackpoint.getLocation());
054  }
055
056  /**
057  * Setzt die Größe des Markers
058  * @param size neue Größe des Markers
059  */
060  public void setSize(int size){
061    this.size = size;
062  }
063
064
065  /**
066  * Setzt die Transparenz des Markers
067  * @param trans Transparenzwert
068  */
069  public void setTransparency(int trans){
070  transparency = trans;
071  }
072
073  /**
074  * Setzt die Farbe des Markers
075  * @param hsb_h Farbwert
076  * @param hsb_s Sättigungswert
077  * @param hsb_b Helligkeitswert
078  */
079  public void setColor(int hsb_h, int hsb_s, int hsb_b){
080    this.hsb_h =hsb_h;
081    this.hsb_s = hsb_s;
082    this.hsb_b = hsb_b;
083  }
084
085  /**
086  * Setzt Farbe des Markers
087  * @param colorstr Farbe
088  */
089  public void setColor(String colorstr){
090    colorstr.toLowerCase();
091  if (colorstr.equals("rot")){
092    hsb_h = 0;
093    hsb_s = 99;
094    hsb_b = 99;
095  } else  if (colorstr.equals("blau")){
096    hsb_h = 240;
097    hsb_s = 99;
098    hsb_b = 99;
099  } else if (colorstr.equals("grün")){
100    hsb_h = 100;
101    hsb_s = 99;
102    hsb_b = 99;
103  } else if (colorstr.equals("gelb")){
104    hsb_h = 60;
105    hsb_s = 99;
106    hsb_b = 99;
107  } else if (colorstr.equals("grau")){
108    hsb_h = 0;
109    hsb_s = 1;
110    hsb_b = 60;
111  }else{
112  System.out.println("Die Farbe ist nicht in der Liste! Farbe mit HSB Codierung möglich!");
113  }
114
115  }
116
117
118
119  /**
120   * Gibt Label aus
121   * @return Label
122   */
123  public String getLabel() {
124    return this.label;
125  }
126
127  /**
128   * Setzt Label
129   * @param label
130   */
131  public void setLabel(String label) {
132    this.label = label;
133  }
134
135  /**
136   * Setzt Textgroesse
137   * @param size Größe
138   */
139  public void setTextSize(int size){
140    this.textsize = size;
141  }
142
143  /**
144   * Textgroesse auslesen
145   * @return Textgröße
146   */
147  public int getTextSize(){
148    return this.textsize;
149  }
150
151  /**
152  * Zeichnet Marker
153  * @param pg Objekt das gezeichnet werden soll
154  * @param x X-Koordinate
155  * @param y Y-Koordinate
156  */
157  public void draw(PGraphics pg, float x, float y){
158    if (!this.isHidden()){
159      pg.pushStyle();
160        pg.noStroke();  // kein Rand
161        pg.fill(hsb_h, hsb_s, hsb_b, transparency);  // Farbe sowie sichtbarkeit
162        pg.ellipse(x, y, size, size);  // Form
163        pg.popStyle();
164    }
165  }
166
167}
